246   public boolean betweenTwoLines(Entity entity)
247   {
248     return 
249     (entity.position.y < (coefDirecteur*entity.position.x + max_ordonneeOrigine)  &&
250      entity.position.y > (coefDirecteur*entity.position.x + min_ordonneeOrigine)
