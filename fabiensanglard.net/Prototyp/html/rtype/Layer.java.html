22 package rtype;
23 
24 import java.util.ArrayList;
25 
26 import rtype.entity.Entity;
27 
28 public class Layer
29 {
30   
31   @SuppressWarnings("unchecked")
32   public ArrayList<Entity> entities = new ArrayList();
33   
34   public Layer(){}
35   
36   public void add(Entity e)
37   {
38     entities.add(e);
39   }
40   
41   public void remove(Entity e)
42   {
43     entities.remove(e);
44   }
45   
46   public void render()
47   {
48     //Entity e = null;
49     
50     for (int i=0 ; i < entities.size() ; i++)
51       {
52         entities.get(i).draw();
53       }
54   }
55 
56   public void update()
57   {
58     Entity e = null;
59     
60     for (int i=0 ; i < entities.size() ; i++)
61       {
62         e = entities.get(i);
63         e.updateTick();
64         e.update();
65         
66       }
67   }
68 }
