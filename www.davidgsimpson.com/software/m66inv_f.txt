      PROGRAM M66INV_MAIN

      INTEGER I, J
      DOUBLE PRECISION MAT(6,6), MATINV(6,6), IMAT(6,6)
      LOGICAL OKFLAG

      DATA MAT / 0.11D0, 4.23D0, 2.22D0, 1.23D0, 3.14D0, 0.21D0,
     $           0.55D0, 0.33D0, 6.01D0, 5.55D0, 0.36D0, 0.09D0,
     $           3.11D0, 2.34D0, 1.23D0, 6.67D0, 0.41D0, 1.99D0,
     $           4.44D0, 1.01D0, 0.75D0, 0.20D0, 3.99D0, 4.75D0,
     $           2.37D0, 5.55D0, 6.14D0, 4.87D0, 3.21D0, 0.76D0,
     $           0.41D0, 0.50D0, 2.74D0, 3.82D0, 1.11D0, 2.78D0 /

C-----------------------------------------------------------------------

C
C     Invert the input matrix.
C

      CALL M66INV (MAT, MATINV, OKFLAG)

C
C     Print the results:
C       original matrix
C       inverse matrix
C       A * Ainv
C       Ainv * A
C

      IF (OKFLAG) THEN
         WRITE (UNIT=*, FMT='(/A/)') ' Matrix:  A'
         WRITE (UNIT=*, FMT='(6F20.10)') ((MAT(I,J), J=1,6), I=1,6)

         WRITE (UNIT=*, FMT='(/A/)') ' Inverse:  Ainv'
         WRITE (UNIT=*, FMT='(6F20.10)') ((MATINV(I,J), J=1,6), I=1,6)

         CALL M66MUL(MAT, MATINV, IMAT)
         WRITE (UNIT=*, FMT='(/A/)') ' Identity 1:  A * Ainv'
         WRITE (UNIT=*, FMT='(6F20.10)') ((IMAT(I,J), J=1,6), I=1,6)

         CALL M66MUL(MATINV, MAT, IMAT)
         WRITE (UNIT=*, FMT='(/A/)') ' Identity 2:  Ainv * A'
         WRITE (UNIT=*, FMT='(6F20.10)') ((IMAT(I,J), J=1,6), I=1,6)
      ELSE
         WRITE (UNIT=*, FMT='(/A)') ' Singular matrix.'
      END IF

      STOP

      END



C***********************************************************************
C M66MUL  -  Mulitply two 6x5 matrices
C***********************************************************************

      SUBROUTINE M66MUL (A, B, AB)

      DOUBLE PRECISION A(6,6), B(6,6), AB(6,6)
      INTEGER I, J, K

      DO I = 1, 6
         DO J = 1, 6
            AB(I,J) = 0.0D0
            DO K = 1, 6
            AB(I,J) = AB(I,J) + A(I,K)*B(K,J)
            END DO
         END DO
      END DO

      RETURN

      END
